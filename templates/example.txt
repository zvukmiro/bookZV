HOW TWO different functions are rendering the SAME template passing different variables into it! 

from flask import Flask, render_templete

app=Flask(__name__)


@app.route('/')
def index():
    return render_template('index.html')

@app.route('greet/<name>')
def greet(name):
    return render_template('index.html', nameTxt=name)

if __name__ == '__main__':
    app.run()
